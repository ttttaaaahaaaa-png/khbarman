        'meta_title',
        'meta_description',
        'meta_keywords',
        'og_title',
        'og_description',
        'og_image',
        'twitter_title',

        'change_frequency',
        'noindex',

    ];

    protected $casts = [
        'meta_keywords' => 'array',
        'structured_data' => 'array',
        'noindex' => 'boolean',
        'nofollow' => 'boolean',
        'priority' => 'float'
    ];

    public function seoable()
    {
        return $this->morphTo();
    }

    // Accessor Ù‡Ø§
    public function getRobotsAttribute($value)
    {
        if ($value) {
            return $value;
        }

        $robots = [];
        
        if ($this->noindex) {
            $robots[] = 'noindex';
        } else {
            $robots[] = 'index';
        }
        
        if ($this->nofollow) {
            $robots[] = 'nofollow';
        } else {
            $robots[] = 'follow';
        }
        
        return implode(', ', $robots);
    }

    // Ù…ØªØ¯Ù‡Ø§ÛŒ Ú©Ù…Ú©ÛŒ
    public function getTitleLength()
    {
        return strlen($this->meta_title ?? '');
    }

    public function getDescriptionLength()
    {
        return strlen($this->meta_description ?? '');
    }

    public function isTitleOptimal()
    {
        $length = $this->getTitleLength();
        return $length >= 30 && $length <= 60;
    }

    public function isDescriptionOptimal()
    {
        $length = $this->getDescriptionLength();
        return $length >= 70 && $length <= 160;
    }

    public function getSeoScore()
    {
        $score = 0;
        
        if ($this->isTitleOptimal()) $score += 30;
        if ($this->isDescriptionOptimal()) $score += 30;
        if ($this->meta_keywords && count($this->meta_keywords) > 0) $score += 20;
        if ($this->og_image) $score += 20;
        
        return $score;
    }
}
